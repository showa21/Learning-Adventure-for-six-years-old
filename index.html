        // Math functions

        function generateMathProblem() {

            const num1 = Math.floor(Math.random() * 20) + 1;

            const num2 = Math.floor(Math.random() * 20) + 1;

            const operations = ['+', '-', '×'];

            const operation = operations[Math.floor(Math.random() * operations.length)];

           

            let problem, answer;

            if (operation === '+') {

                problem = `${num1} + ${num2} = ?`;

                answer = num1 + num2;

            } else if (operation === '-') {

                const bigger = Math.max(num1, num2);

                const smaller = Math.min(num1, num2);

                problem = `${bigger} - ${smaller} = ?`;

                answer = bigger - smaller;

            } else {

                const small1 = Math.floor(Math.random() * 10) + 1;

                const small2 = Math.floor(Math.random() * 10) + 1;

                problem = `${small1} × ${small2} = ?`;

                answer = small1 * small2;

            }

           

            document.getElementById('mathProblem').textContent = problem;

            currentMathAnswer = answer;

            document.getElementById('mathAnswer').value = '';

        }

 

        function checkMathAnswer() {

            const userAnswer = parseInt(document.getElementById('mathAnswer').value);

            if (userAnswer === currentMathAnswer) {

                showFeedback(true, "Correct! 🎉");

                mathCorrect++;

                updateProgress('mathProgress', mathCorrect, 10);

                setTimeout(generateMathProblem, 1500);

            } else {

                showFeedback(false, "Try again! 💪");

            }

        }
